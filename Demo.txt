#!/bin/bash

# Check if input.txt exists
if [ ! -f "input.txt" ]; then
    echo "Error: input.txt file not found!"
    exit 1
fi

# Read each line from input.txt
while IFS= read -r cluster_name; do
    # Skip empty lines
    [ -z "$cluster_name" ] && continue
    
    # Define the output filename
    output_file="servicemonitor-${cluster_name}.yaml"
    
    # Create the servicemonitor YAML content
    cat << EOF > "$output_file"
apiVersion: monitoring.coreos.com/v1
kind: ServiceMonitor
metadata:
  name: ${cluster_name}-monitor
  namespace: monitoring
spec:
  endpoints:
  - port: https
    scheme: https
    path: /metrics
    targetPort: 6443
    interval: 30s
    tlsConfig:
      insecureSkipVerify: true
    relabelings:
    - sourceLabels: [__address__]
      targetLabel: __address__
      replacement: api.${cluster_name}.ebiz.com:6443
  - port: https
    scheme: https
    path: /metrics
    targetPort: 443
    interval: 30s
    tlsConfig:
      insecureSkipVerify: true
    relabelings:
    - sourceLabels: [__address__]
      targetLabel: __address__
      replacement: j14v.apps.${cluster_name}.ebiz.com:443
  selector:
    matchLabels:
      app: ${cluster_name}-service
EOF

    echo "Created $output_file"
done < "input.txt"

echo "ServiceMonitor file generation completed!"
